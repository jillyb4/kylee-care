#!/bin/bash # ───────────────────────────────────────────────────────────────── # push-to-github.sh # Copies the updated kylee-print.html into the local kylee-care # repo and pushes it to GitHub. Run this after any schedule update. # # SETUP (one time only): # 1. Set REPO_PATH below to wherever you cloned jillyb4/kylee-care # 2. Make this script executable: chmod +x push-to-github.sh # 3. Run it: ./push-to-github.sh # ───────────────────────────────────────────────────────────────── # ── EDIT THIS ──────────────────────────────────────────────────── REPO_PATH="$HOME/kylee-care" # path to your local git repo SOURCE_FILE="kylee-print.html" # the file we update each time DEST_FILE="kylee-print.html" # filename in the repo (same) # ───────────────────────────────────────────────────────────────── # Get the directory this script lives in so source path is relative SCRIPT_DIR="$(cd "$(dirname "$0")" && pwd)" echo "→ Copying updated file to repo..." cp "$SCRIPT_DIR/$SOURCE_FILE" "$REPO_PATH/$DEST_FILE" echo "→ Staging changes..." cd "$REPO_PATH" || { echo "ERROR: Repo not found at $REPO_PATH"; exit 1; } git add "$DEST_FILE" # Auto-generate commit message with today's date COMMIT_MSG="Update nursing schedule $(date '+%Y-%m-%d')" echo "→ Committing: $COMMIT_MSG" git commit -m "$COMMIT_MSG" echo "→ Pushing to GitHub..." git push echo "" echo "✅ Done! Site will update at https://jillyb4.github.io/kylee-care/"
